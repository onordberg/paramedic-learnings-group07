# Clippy AI Assistant — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an authentic animated Clippy character to the Paramedic Learnings app as a context-aware AI assistant backed by Claude, toggled via a status bar icon.

**Architecture:** A `ClippyProvider` React Context in the root layout holds open state, message history, and page context. A floating `ClippyWidget` renders the real `clippyjs` character with a Win 3.11-style chat panel above it. An API route streams Claude responses using `claude-haiku-4-5-20251001`.

**Tech Stack:** Next.js 16 App Router, React 19, `clippyjs` (Office 97 sprite animations), `anthropic` SDK, Vitest + React Testing Library.

export type TopicData = {
  title: string;
  summary: string;
  guidance: string;
};

export function buildPageContext(page: string, topic?: TopicData): string {
  if (page === "topic-detail") {
    if (!topic) return "User is reading a topic.";
    return `User is reading the topic "${topic.title}". Summary: ${topic.summary}. Guidance: ${topic.guidance}`;
  }
  if (page === "topic-new") return "User is creating a new topic.";
  if (page === "topic-list") return "User is browsing the topic list.";
  return "User is on the Paramedic Learnings platform.";
}
```

"use client";

import React, { createContext, useContext, useState } from "react";

export type ClippyMessage = {
  role: "user" | "assistant";
  content: string;
};

type ClippyContextValue = {
  isOpen: boolean;
  setIsOpen: (v: boolean) => void;
  messages: ClippyMessage[];
  setMessages: React.Dispatch<React.SetStateAction<ClippyMessage[]>>;
  pageContext: string;
  setPageContext: (v: string) => void;
};

const ClippyContext = createContext<ClippyContextValue | null>(null);

export function ClippyProvider({ children }: { children: React.ReactNode }) {
  const [isOpen, setIsOpen] = useState(false);
  const [messages, setMessages] = useState<ClippyMessage[]>([]);
  const [pageContext, setPageContext] = useState(
    "User is on the Paramedic Learnings platform."
  );

  return (
    <ClippyContext.Provider
      value={{ isOpen, setIsOpen, messages, setMessages, pageContext, setPageContext }}
    >
      {children}
    </ClippyContext.Provider>
  );
}

export function useClippy() {
  const ctx = useContext(ClippyContext);
  if (!ctx) throw new Error("useClippy must be used inside ClippyProvider");
  return ctx;
}
```

## Task 4: `/api/clippy` streaming route

**Files:**
- Create: `src/app/api/clippy/route.ts`

- [ ] **Step 1: Create the API route**

Create `src/app/api/clippy/route.ts`:

```ts
import { NextRequest } from "next/server";
import Anthropic from "@anthropic-ai/sdk";

const anthropic = new Anthropic();

const SYSTEM_PROMPT = `You are Clippy, the helpful assistant from Microsoft Office, now embedded in Paramedic Learnings — a knowledge platform for ambulance personnel.

You have Clippy's cheerful, slightly over-eager personality. Use his classic openers occasionally ("It looks like you're...", "Did you know..."), but keep all medical and operational advice accurate and practical. Never invent clinical facts.

Current context: CONTEXT_PLACEHOLDER

Keep responses concise (2–4 sentences unless more detail is genuinely needed).`;

export async function POST(req: NextRequest) {
  const { messages, pageContext } = (await req.json()) as {
    messages: { role: "user" | "assistant"; content: string }[];
    pageContext: string;
  };

  const systemPrompt = SYSTEM_PROMPT.replace("CONTEXT_PLACEHOLDER", pageContext);

  const stream = anthropic.messages.stream({
    model: "claude-haiku-4-5-20251001",
    max_tokens: 1024,
    system: systemPrompt,
    messages,
  });

  const encoder = new TextEncoder();
  const readable = new ReadableStream({
    async start(controller) {
      try {
        for await (const chunk of stream) {
          if (
            chunk.type === "content_block_delta" &&
            chunk.delta.type === "text_delta"
          ) {
            controller.enqueue(encoder.encode(chunk.delta.text));
          }
        }
      } finally {
        controller.close();
      }
    },
  });

  return new Response(readable, {
    headers: { "Content-Type": "text/plain; charset=utf-8" },
  });
}
```

Create `src/app/_components/ClippyWidget.tsx`:

```tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { useClippy, type ClippyMessage } from "./ClippyProvider";

type ClippyAgent = {
  show(): void;
  hide(): void;
  speak(text: string): void;
  play(animation: string): void;
  animate(): void;
  stop(): void;
};

export function ClippyWidget() {
  const { isOpen, setIsOpen, messages, setMessages, pageContext } = useClippy();
  const agentRef = useRef<ClippyAgent | null>(null);
  const [input, setInput] = useState("");
  const [isStreaming, setIsStreaming] = useState(false);
  const messagesEndRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    let mounted = true;
    import("clippyjs").then(({ default: clippy }) => {
      if (!mounted) return;
      clippy.load(
        "Clippy",
        (agent: ClippyAgent) => {
          if (!mounted) return;
          agentRef.current = agent;
        },
        "/clippy-agents/"
      );
    });
    return () => {
      mounted = false;
    };
  }, []);

  useEffect(() => {
    const agent = agentRef.current;
    if (!agent) return;
    if (isOpen) {
      agent.show();
      agent.play("Greeting");
    } else {
      agent.hide();
    }
  }, [isOpen]);

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  async function handleSend() {
    if (!input.trim() || isStreaming) return;

    const userMessage: ClippyMessage = { role: "user", content: input.trim() };
    const nextMessages = [...messages, userMessage];
    setMessages(nextMessages);
    setInput("");
    setIsStreaming(true);
    agentRef.current?.play("Thinking");

    try {
      const res = await fetch("/api/clippy", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ messages: nextMessages, pageContext }),
      });

      if (!res.ok || !res.body) throw new Error("Request failed");

      const reader = res.body.getReader();
      const decoder = new TextDecoder();
      let assistantText = "";

      setMessages((prev) => [...prev, { role: "assistant", content: "" }]);

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        assistantText += decoder.decode(value, { stream: true });
        setMessages((prev) => {
          const updated = [...prev];
          updated[updated.length - 1] = {
            role: "assistant",
            content: assistantText,
          };
          return updated;
        });
      }

      agentRef.current?.play("Writing");
    } catch {
      setMessages((prev) => [
        ...prev,
        {
          role: "assistant",
          content: "Hmm, something seems to have gone wrong. Want to try again?",
        },
      ]);
      agentRef.current?.play("GetAttention");
    } finally {
      setIsStreaming(false);
    }
  }

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  }

  if (!isOpen) return null;

  return (
    <div
      style={{
        position: "absolute",
        bottom: "28px",
        right: "8px",
        width: "260px",
        zIndex: 50,
        display: "flex",
        flexDirection: "column",
        alignItems: "flex-end",
        gap: "4px",
      }}
    >
      {/* Chat panel */}
      <div
        className="win-raised"
        style={{ width: "100%", background: "#c0c0c0" }}
      >
        {/* Title bar */}
        <div className="win-titlebar">
          <span style={{ display: "flex", alignItems: "center", gap: "4px" }}>
            📎 Clippy
          </span>
          <button
            className="win-titlebar-btn"
            onClick={() => setIsOpen(false)}
            aria-label="Close Clippy"
            style={{ fontSize: "9px" }}
          >
            ✕
          </button>
        </div>

        {/* Message history */}
        <div
          className="win-sunken-deep"
          style={{
            margin: "4px",
            background: "#ffffff",
            maxHeight: "200px",
            overflowY: "auto",
            padding: "4px",
            display: "flex",
            flexDirection: "column",
            gap: "4px",
          }}
        >
          {messages.length === 0 && (
            <p style={{ color: "#808080", fontSize: "10px", margin: 0 }}>
              It looks like you could use some help! Ask me anything.
            </p>
          )}
          {messages.map((msg, i) => (
            <div
              key={i}
              style={{
                alignSelf: msg.role === "user" ? "flex-end" : "flex-start",
                background: msg.role === "user" ? "#000080" : "#ffffc0",
                color: msg.role === "user" ? "#ffffff" : "#000000",
                border: msg.role === "assistant" ? "1px solid #808080" : "none",
                padding: "3px 8px",
                fontSize: "10px",
                maxWidth: "90%",
                lineHeight: "1.4",
              }}
            >
              {msg.content}
            </div>
          ))}
          <div ref={messagesEndRef} />
        </div>

        {/* Input row */}
        <div
          style={{
            display: "flex",
            gap: "4px",
            padding: "0 4px 4px 4px",
          }}
        >
          <input
            className="win-input"
            type="text"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={handleKeyDown}
            disabled={isStreaming}
            placeholder="Ask Clippy..."
            aria-label="Ask Clippy"
            style={{ flex: 1, fontSize: "10px" }}
          />
          <button
            className="win-btn"
            onClick={handleSend}
            disabled={isStreaming || !input.trim()}
            aria-label="Send"
            style={{ minWidth: 0, padding: "2px 8px", fontSize: "10px" }}
          >
            Send
          </button>
        </div>
      </div>
    </div>
  );
}
```
